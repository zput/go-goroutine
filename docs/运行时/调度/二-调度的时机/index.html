    中断恢复goroutine #

    猜猜这个程序的运行结果

    package main
import (
    "fmt"
    "runtime"
)
func g1() {
    sum := -1
    for {
        sum++
    }
}
func main() {
    runtime.GOMAXPROCS(0)
    go g1()
    for {
        runtime.Gosched() //如果注释掉这一行，将会在开始时候打印，后面不会再打印
        fmt.Println("main is scheduled!")
    }
}
    • 将不会打印[“main is scheduled!”]
